Insta-wrapper Example





// when the user presses the 'Go!' button:

function go() {
  var urn = $('#urn').val();

  var request = $.ajax({
    url: '/api' + urn,
    method: 'GET'
  });

  request.done(function(response) {
    if (response && response.data && response.data.length) {
      for (var i=0; i').load(function() {
          $(this).width(300).height(300).appendTo('#images');
        });
      }
    }
  });
}
